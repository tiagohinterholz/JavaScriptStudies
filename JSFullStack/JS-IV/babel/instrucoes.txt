npm init -y

npm install --save-dev @babel/core @babel/cli @babel/preset-env

npx babel index.js - coloca no terminal o codigo alterado

npx babel index.js --out-file dist.js - cria um novo arquivo alterado

npx babel index.js --out-file dist.js --presets=@babel/preset-env - chama o preset especifico que quer o file.js.

npx babel src --out-dir dist --presets=@babel/preset-env - criada a pasta src passa ela como referencia e todos file.js dentro se convertem
